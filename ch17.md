# Iterator

- 如果一个对象有 [Symbol.iterator] 方法, 则实现了 Iterable 接口 (鸭子类型)
- 实现了 Iterable 接口的对象可以使用 for...of 遍历

```ts
interface Iterable {
  [Symbol.iterator](): Iterator;
}

interface Iterator {
  // 指针
  next(value?: any): IterationResult; // 移动指针
}

interface IterationResult {
  value: any;
  done: boolean; // 遍历是否结束
}
```

```js
let obj = {
  [Symbol.iterator]: function () {
    return /* Iterator */ {
      next: function () {
        return /* IteratorResult */ {
          value: 1,
          done: false,
        };
      },
    };
  },
};
for (let val of obj) {
  console.log(val); // 1 1 1 ...
}
```

原生实现了 Iterable 接口的对象/函数

- Array
- Map (Object 没有实现 Iterable 接口)
- Set
- String
- 函数的 arguments 对象
- NodeList 对象

```js
let arr = [1, 2, 3];
let iter = arr[Symbol.iterator]();
console.log(iter.next()); // { value: 1, done: false }
console.log(iter.next()); // { value: 2, done: false }
console.log(iter.next()); // { value: 3, done: false }
console.log(iter.next()); // { value: undefined, done: true }
```

使用 Iterator 实现链表

```js
class RangeIter {
  constructor(val, end) {
    this.val = val;
    this.end = end;
  }

  [Symbol.iterator]() {
    return /* Iterator */ this;
  }

  next() {
    let val = this.val;
    if (val < this.end) {
      this.val++;
      return /* IteratorResult */ { done: false, value: val };
    }
    return /* IteratorResult */ { done: true, value: undefined };
  }
}

for (let val of new RangeIter(0, 3)) {
  console.log(val); // 0 1 2
}
```

```js
function Node(val) {
  this.val = val;
  this.sibling = null;
}

Node.prototype[Symbol.iterator] = function () {
  let cur = this;
  return /* Iterator */ {
    next: () => {
      // console.log(this); // first, first, first
      // console.log(cur); // first, second, third
      if (cur) {
        let val = cur.val;
        cur = cur.sibling;
        return { done: false, value: val };
      }
      return { done: true /* , value: undefined */ };
    },
  };
};

let first = new Node(1);
let second = new Node(2);
let third = new Node(3);
first.sibling = second;
second.sibling = third;

for (let nodeVal of first) {
  console.log(nodeVal);
}
```

实现可 for...of 遍历的对象

```js
let obj = {
  data: ["foo", "bar", "baz"],
  [Symbol.iterator]() {
    let self = this;
    let idx = 0;
    return /* Iterator */ {
      next() {
        if (idx < self.data.length) {
          return /* IteratorResult */ {
            value: self.data[idx++],
            done: false,
          };
        }
        return { value: self.data[idx], done: true };
      },
    };
  },
};

for (let item of obj) {
  console.log(item);
}
```

对于类似数组的对象: 有数值键 (键字符串可以转换为数值) 和 length 属性, 则实现 Iterable 接口的简单方法是, 直接引用数组的 [Symbol.iterator] 方法

```js
NodeList.prototype[Symbol.iterator] === Array.prototype[Symbol.iterator]; // true
NodeList.prototype[Symbol.iterator] === new Array()[Symbol.iterator]; // true
NodeList.prototype[Symbol.iterator] === [][Symbol.iterator]; // true
```

```js
let obj /* Iterable */ = {
  0: "foo",
  1: "bar",
  2: "baz",
  length: 3,
  [Symbol.iterator]: Array.prototype[Symbol.iterator],
};
for (let item of obj) {
  console.log(item); // foo bar baz
}
```

### 调用 Iterator 接口的场景

1. 解构赋值
2. 扩展运算符 ...
3. yield\* + 可遍历对象
4. for...of, Array.from(), Map(), Set(), WeakMap(), WeakSet(), Promise.all(), Promise.race(), ...

```js
let makeGen /* GeneratorFunction */ = function* () {
  yield 1;
  yield* [2, 3, 4]; // 委托
  yield 5;
};
let gen /* Generator */ = makeGen();
for (let i = 0; i < 6; i++) {
  console.log(gen.next());
}
```

### String 也实现了 Iterable 接口

```js
let str = "bar"; // 基本数据类型
// 隐式调用 String 对象包装器
console.log(typeof str[Symbol.iterator]); // function
let iter /* Iterator */ = str[Symbol.iterator]();

for (let i = 0; i < 4; i++) {
  console.log(iter.next());
}
```

覆盖原生的 [Symbol.iterator] 方法

```js
let str = new String("baz");
console.log([...str]); // [ "b", "a", "z" ]
// 覆盖原生的 Symbol.iterator 方法
str[Symbol.iterator] = function () {
  return /* Iterator */ {
    next() {
      if (this._) {
        this._ = false;
        return { value: "bye", done: false };
      }
      return { done: true };
    },
    _: true,
  };
};
console.log([...str]); // [ 'bye' ]
```

### Iterator 接口和 Generator

```js
function* makeGen() {
  yield* [1, 2, 3];
} // GeneratorFunction

let gen = makeGen(); // Generator, Iterable
console.log([...gen]); // [1, 2, 3]

let foo = {
  [Symbol.iterator]: function* () {
    yield* [1, 2, 3];
  },
};
console.log([...foo]); // [1, 2, 3]

let bar = {
  *[Symbol.iterator]() {
    yield* [1, 2, 3];
  },
};
console.log([...bar]); // [1, 2, 3]
```
