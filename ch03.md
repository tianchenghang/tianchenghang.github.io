# 变量的解构赋值

### 数组的解构赋值

```js
let [foo, [[bar], baz]] = [1, [[2], 3]];
console.log(foo); // 1
console.log(bar); // 2
console.log(baz); // 3

let [, , third] = ["foo", "bar", "baz"];
console.log(third); // "baz"

let [x, , y] = [1, 2, 3];
console.log(x); // 1
console.log(y); // 3

let [x1, ...y1] = [1, 2, 3, 4];
console.log(x1); // 1
console.log(y1); // [2, 3, 4]

let [x2, y2, ...z2] = ["a"];
console.log(x2); // "a"
console.log(y2); // undefined
console.log(z2); // []
```

如果解构失败, 则左值为 undefined

```js
let [foo] = [];
console.log(foo); // undefined

let [bar, baz] = [1];
console.log(baz); // undefined
```

不完全解构

```js
let [x, y] = [1, 2, 3];
console.log(x); // 1
console.log(y); // 2

let [a, [b], c] = [1, [2, 3], 4];
console.log(a); // 1
console.log(b); // 2
console.log(c); // 4
```

如果右值不可遍历, 则抛出错误

```js
let [foo] = {}; // TypeError: {} is not iterable
```

所有实现了 Iterator 接口的对象, 都可以使用解构赋值

```js
let [x, y, z] = new Set(["a", "b", "c"]);
console.log(x, y, z); // a b c

function* fibs() {
  let curr = 0;
  let next = 1;
  while (true) {
    yield curr;
    [curr, next] = [next, curr + next];
  }
}

let [first, second, third, fourth, fifth] = fibs();
console.log(first, second, third, fourth, fifth); // 0 1 1 2 3
```

解构赋值可以指定默认值

```js
let [foo = true] = [];
console.log(foo); // true

let [x, y = "b"] = ["a"];
console.log(x, y); // a b

let [x1, y1 = "b"] = ["a", undefined];
console.log(x1, y1); // a b

let [x2 = 1] = [undefined];
console.log(x2); // 1

let [x3 = 1] = [null];
console.log(x3); // null

function f() {
  return "wtf";
}
let [x4 = f()] = [1];
console.log(x4); // 1

let [x5 = 1, y5 = x5] = [];
console.log(x5, y5); // 1 1

let [x6 = 1, y6 = x6] = [2];
console.log(x6, y6); // 2 2

let [x7 = 1, y7 = x7] = [1, 2];
console.log(x7, y7); // 1 2

let [x8 = y8, y8 = 1] = [];
console.log(x8, y8); // ReferenceError: y8 is not defined
```

- undefined: 未被赋值
- null: 已被赋值, 赋空值

### 对象的解构赋值

```js
let { foo, bar } = { foo: "aaa", bar: "bbb" };
console.log(foo); // "aaa"
console.log(bar); // "bbb"
```

- 数组按下标解构
- 对象按属性名解构

```js
let { baz } = { foo: "aaa", bar: "bbb" };
console.log(baz); // undefined

let { baz } = { foo: "aaa", bar: "bbb" };
console.log(baz); // undefined
```

对象的解构赋值是以下的简写

```js
let obj = { foo: "aaa", bar: "bbb" };
// let { foo, bar } = { foo: "aaa", bar: "bbb" };

// 对象的解构赋值是以下的简写
let { foo /* prop */: foo /* var */, bar /* prop */: bar /* var */ } = obj;
console.log(foo, bar); // aaa bbb

// 起 "别名"
let { foo /* prop */: f /* var */, bar /* prop */: b /* var */ } = obj;
console.log(f, b); // aaa bbb

let greet = {
  p: ["hello", { y: "world" }],
};

let {
  p: [x /* 先按下标解构 */, { y /* 再按属性名解构 */ }],
} = greet;
console.log(x); // "hello"
console.log(y); // "world"

let {
  p,
  p: [x1, { y: y1 }],
} = greet;
console.log(p, x1, y1);
// ["hello", { y: "world" }]
// "hello"
// "world"
```

```js
let obj = {};
let arr = [];

({ foo: obj.prop, bar: arr[0] } = { foo: 123, bar: true });

console.log(obj); // {prop:123}
console.log(arr); // [true]
```

可以解构继承的属性

```js
const son = {};
const pa = { foo: "bar" };
Object.setPrototypeOf(son, pa);
console.log(son.__proto__ === pa); // true

const { foo } = son;
console.log(foo); // "bar"
```

解构赋值可以指定默认值

```js
let { x = 3, y } = {};
console.log(x, y); // 3 undefined

let { x1, y1 = 5 } = { x1: 3 };
console.log(x1, y1); // 3 5

let { x: x2 = 3 } = {};
console.log(x2); // 3

let { x: x3 = 3 } = { x: 5 };
console.log(x3); // 5

let { x4 = 3 } = { x4: undefined };
console.log(x4); // 3

let { x5 = 3 } = { x5: null };
console.log(x5); // null
```

坑: v8 将 { x } 解释为一个代码块

```js
let x;
{ x } = { x: 1 }; // SyntaxError
```

解决

```js
let x;
({ x } = { x: 1 });
console.log(x); // 1
```

数组是特殊的对象, 有 length 属性, 可以使用对象的解构赋值解构数组

```js
let arr = [1, 2, 3];
let { 0: first, [1]: second, [arr.length - 1]: third, length: len } = arr;
console.log(first); // 1
console.log(second); // 2
console.log(third); // 3
console.log(len); // 3
```

### 字符串的解构赋值

可以使用数组的解构赋值解构字符串

```js
const [a, b, c] = "hello";
console.log(a, b, c); // h e l

const [x, y, ...z] = "world";
console.log(x, y, z); // w o ['r', 'l', 'd']
```

字符串是特殊的对象, 有 length 属性, 可以使用对象的解构赋值解构字符串

```js
let { length: len } = "hello";
console.log(len); // 5
```

### 数值和布尔值的解构赋值

如果右值为基本类型, 则会转换为对象

```js
// 123 自动装箱为 new String(123)
let { toString: toStr1 } = 123;
console.log(toStr1 === Number.prototype.toString); // true

// true 自动装箱为 new Boolean(true)
let { toString: toStr2 } = true;
console.log(toStr2 === Boolean.prototype.toString); // true
```

undefined, null 不能转换为对象, 不能解构赋值

```js
let { prop: x } = undefined; // TypeError
let { prop: y } = null; // TypeError
```

### 函数参数的解构赋值

```js
// 传递参数时, 使用数组的解构赋值
function add([x, y]) {
  return x + y;
}

console.log(add([1, 2])); // 3
```

```js
// 传递参数时, 使用对象的解构赋值
function move({ x = 0, y = 0 } = {} /* {} 参数默认值 */) {
  return [x, y];
}
console.log(move({ x: 3, y: 8 })); // [3, 8]
console.log(move({ x: 3 })); // [3, 0]
console.log(move({})); // [0, 0]
console.log(move()); // [0, 0]

// 传递参数时, 使用对象的解构赋值
function move1({ x, y } = { x: 0, y: 0 }) {
  return [x, y];
}
console.log(move1({ x: 3, y: 8 })); // [3, 8]
console.log(move1({ x: 3 })); // [3, undefined]
console.log(move1({})); // [undefined, undefined]
console.log(move1()); // [0, 0]
```

```js
// [1, 'wtf', 3]
console.log([1, undefined, 3].map((x = "wtf") => x));
// [1, null, 3]
console.log([1, null, 3].map((x = "wtf") => x));
```

### Demo

swap

```js
let x = 1;

```

函数返回多个值

```js
function retarr() {
  return [1, 2, 3];
}
let [a, b, c] = retarr();

function retobj() {
  return {
    foo: 1,
    bar: 2,
  };
}
let { foo, bar } = retobj();
```

传递参数时解构赋值: 略

解析 JSON 数据

```js
let resp = {
  id: 1,
  status: "OK",
  data: [4, 16],
};

let { id, status, data: nums } = resp;
console.log(id, status, nums); // 1 OK [4, 16]
```

函数参数默认值: 略

遍历 map

```js
const kvs = new Map([
  ["first", "hello"],
  ["second", "world"],
]);

for (let [k, v] of kvs) {
  console.log(k, "=>", v);
}
```

```js
const { item1, item2 } = require("mod");
```
