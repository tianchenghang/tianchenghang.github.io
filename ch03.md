# 解构赋值

### 数组的解构赋值

```js
let [foo, [[bar], baz]] = [1, [[2], 3]];
console.log(foo); // 1
console.log(bar); // 2
console.log(baz); // 3

let [, , third] = ["foo", "bar", "baz"];
console.log(third); // "baz"

let [first, , last] = [1, 2, 3];
console.log(first); // 1
console.log(last); // 3

let [head, ...tail] = [1, 2, 3, 4];
console.log(head); // 1
console.log(tail); // [2, 3, 4]

let [x, y, ...rest] = ["a"];
console.log(x); // "a"
console.log(y); // undefined
console.log(rest); // []
```

如果解构失败, 则左值为 undefined

```js
let [foo] = [];
console.log(foo); // undefined

let [bar, baz] = [1];
console.log(baz); // undefined
```

不完全解构

```js
let [x, y] = [1, 2, 3];
console.log(x); // 1
console.log(y); // 2

let [a, [b], c] = [1, [2, 3], 4];
console.log(a); // 1
console.log(b); // 2
console.log(c); // 4
```

如果右值不可遍历, 则抛出错误

```js
let [foo] = {}; // TypeError: {} is not iterable
```

所有实现了 Iterator 接口的对象, 都可以使用解构赋值

```js
let [x, y, z] = new Set(["a", "b", "c"]);
console.log(x, y, z); // a b c

function* fibs() {
  let curr = 0;
  let next = 1;
  while (true) {
    yield curr;
    [curr, next] = [next, curr + next];
  }
}

let [first, second, third] = fibs();
console.log(first, second, third); // 0 1 1
```

解构赋值可以指定默认值

```js
let [foo = true] = [];
console.log(foo); // true

let [x, y = "b"] = ["a"];
console.log(x, y); // a b

let [x1, y1 = "b"] = ["a", undefined];
console.log(x1, y1); // a b

let [x2 = 1] = [undefined];
console.log(x2); // 1

let [x3 = 1] = [null];
console.log(x3); // null

function f() {
  return "wtf";
}
let [x4 = f()] = [1];
console.log(x4); // 1

let [x5 = 1, y5 = x5] = [];
console.log(x5, y5); // 1 1

let [x6 = 1, y6 = x6] = [2];
console.log(x6, y6); // 2 2

let [x7 = 1, y7 = x7] = [1, 2];
console.log(x7, y7); // 1 2

let [x8 = y8, y8 = 1] = [];
console.log(x8, y8); // ReferenceError: y8 is not defined
```

- undefined: 未被赋值
- null: 已被赋值, 赋空值

### 对象的解构赋值

```js
let { foo, bar } = { foo: "a", bar: "b" };
console.log(foo, bar); // a b
```

> 数组按下标解构 (数组对象的属性名是下标)
> 对象按属性名解构

```js
let { bar, foo } = { foo: "a", bar: "b" };
console.log(foo, bar); // a b

let { baz } = { foo: "a", bar: "b" };
console.log(baz); // undefined
```

对象的解构赋值是以下的简写

```js
let obj = { foo: "a", bar: "b" };
// let { foo, bar } = { foo: "a", bar: "b" };

// 对象的解构赋值是以下的简写
let { foo /* prop */: foo /* var */, bar /* prop */: bar /* var */ } = obj;
console.log(foo, bar); // a b

// 起 "别名"
let { foo /* prop */: f /* var */, bar /* prop */: b /* var */ } = obj;
console.log(f, b); // a b

let greet = {
  p: ["hello", { y: "world" }],
};

let {
  p: [x /* 先按下标解构 */, { y /* 再按属性名解构 */ }],
} = greet;
console.log(x); // hello
console.log(y); // world

let {
  p,
  p: [x1, { y: y1 }],
} = greet;
console.log(p, x1, y1);
// ["hello", { y: "world" }]
// hello
// world
```

```js
let obj = {};
let arr = [];
({ foo: obj.prop, bar: arr[0] } = { foo: 123, bar: true });
console.log(obj); // { prop: 123 }
console.log(arr); // [ true ]
```

可以解构继承的属性

```js
let ch = {};
let pa = { foo: "bar" };
Object.setPrototypeOf(ch, pa);
console.log(ch.__proto__ === pa); // true
let { foo } = ch;
console.log(foo); // "bar"
```

解构赋值可以指定默认值

```js
let { x = 3 } = {};
console.log(x); // 3

let { x1, y1 = 5 } = { x1: 3 };
console.log(x1, y1); // 3 5

let { x: x2 = 3 } = {};
console.log(x2); // 3

let { x: x3 = 3 } = { x: 5 };
console.log(x3); // 5

let { x4 = 3 } = { x4: undefined };
console.log(x4); // 3

let { x5 = 3 } = { x5: null };
console.log(x5); // null
```

```js
let x;
// { x } = { x: 1 }; // SyntaxError
//  x } 被解释为一个代码块, 需要使用 () 包裹
({ x } = { x: 1 });
console.log(x); // 1
```

数组是特殊的对象, 有 length 属性, 可以使用对象的解构赋值解构数组

```js
let arr = [1, 2, 3];
let { 0: first, [1]: second, [arr.length - 1]: third, length: len } = arr;
// [expr] 属性名表达式
console.log(first); // 1
console.log(second); // 2
console.log(third); // 3
console.log(len); // 3
```

### 字符串的解构赋值

可以使用数组的解构赋值解构字符串

```js
let [a, b, c] = "hello";
console.log(a, b, c); // h e l

let [x, y, ...z] = "world";
console.log(x, y, z); // w o ['r', 'l', 'd']

let { length: len } = "hello";
console.log(len); // 5
```

### 数值和布尔值的解构赋值

如果右值为基本类型, 则会转换为对象

```js
// 123 自动包装为 new Number(123)
let { toString: toStr1 } = 123;
console.log(toStr1 === Number.prototype.toString); // true

// true 自动包装为 new Boolean(true)
let { toString: toStr2 } = true;
console.log(toStr2 === Boolean.prototype.toString); // true
```

undefined, null 不能转换为对象, 不能解构赋值

```js
let { prop: x } = undefined; // TypeError
let { prop: y } = null; // TypeError
```

### 函数参数的解构赋值

```js
// 传递参数时, 使用数组的解构赋值
function add([x, y]) {
  return x + y;
}
console.log(add([1, 2])); // 3

[
  [1, 2],
  [3, 4],
].map(([a, b] /* item */) => a + b); // [ 3, 7 ]
```

```js
// 传递参数时, 使用对象的解构赋值
function move1({ x = 0, y = 0 } = {} /* {} 参数默认值 */) {
  return [x, y];
}
console.log(move1({ x: 1, y: 2 })); // [1, 2]
console.log(move1({ x: 1 })); // [1, 0]
console.log(move1({})); // [0, 0]
console.log(move1()); // [0, 0]

// 传递参数时, 使用对象的解构赋值
function move2({ x, y } = { x: 0, y: 0 }) {
  return [x, y];
}
console.log(move2({ x: 1, y: 2 })); // [1, 2]
console.log(move2({ x: 1 })); // [1, undefined]
console.log(move2({})); // [undefined, undefined]
console.log(move2()); // [0, 0]
```

```js
// [1, 'wtf', 3]
console.log([1, undefined, 3].map((x = "wtf") => x));
// [1, null, 3]
console.log([1, null, 3].map((x = "wtf") => x));
```

### Demo

交换

```js
let x = 1,
  y = 2;
[x, y] = [y, x];
```

函数返回多个值

```js
function retArr() {
  return [1, 2, 3];
}
let [a, b, c] = retArr();

function retObj() {
  return {
    foo: 1,
    bar: 2,
  };
}
let { foo, bar } = retObj();
```

解析 JSON 数据

```js
let resp = {
  id: 1,
  status: "OK",
  data: [4, 16],
};

let { id, status, data: nums } = resp;
console.log(id, status, nums); // 1 OK [4, 16]
```

遍历 map

```js
let kvs = new Map([
  ["first", "hello"],
  ["second", "world"],
]);

for (let [k, v] of kvs) {
  console.log(k, "=>", v);
}
```

导入模块

```js
let { item1, item2 } = require("mod");
```
