# 函数

### 参数默认值

```js
function log(x, y = "World") {
  console.log(x, y);
}

log("Hello"); // Hello World
log("Hello", "typescript"); // Hello typescript
log("Hello", ""); // Hello
```

```js
function foo(x) {
  let x = 1; // SyntaxError: Identifier 'x' has already been declared
}

function bar(x = 5) {
  const x = 2; // SyntaxError: Identifier 'x' has already been declared
}
```

> 有默认值的参数, 推荐放在参数列表的尾部

### 函数的 length 属性

length 属性: 返回参数数量 (不统计有默认值的参数和后续参数; 不统计 rest 剩余参数)

```js
// length 属性不统计有默认值的参数和后续参数
(function (a) {}).length; // 1
(function (a = 5) {}).length; // 0
(function (a, b, c = 5, d, e) {}).length; // 2

// length 属性不统计 rest 剩余参数
(function (...args) {}).length; // 0
```

### 作用域 context

例子

```js
let x = 1;
function foo(x = x) {}
foo(); // ReferenceError: Cannot access 'x' before initialization

// function foo(y = x) {}
// foo(); // OK
```

```js
let foo = "outer";
function bar(func = () => foo) {
  let foo = "inner";
  console.log(func());
}
bar(); // outer
```

```js
function bar(func = () => foo) {
  let foo = "inner";
  console.log(func());
}
bar(); // ReferenceError: foo is not defined
```

```js
var x = 1; // global.x = 1
function foo(
  x /* = 666 */,
  y = function () {
    console.log(x); // foo.args.x = undefined
    x = 2; // foo.args.x = 2
    console.log(x); // foo.args.x = 2
  },
) {
  var x = 3; // foo.x = 3
  y();
  console.log(x); // foo.x = 3
}
foo(); // undefined 2 3
console.log(x); // global.x = 1
```

```js
var x = 1; // global.x = 1
function foo(
  x,
  y = function () {
    console.log(x); // foo.args.x = 3
    x = 2; // foo.args.x = 2
    console.log(x); // foo.args.x = 2
  },
) {
  x = 3; // foo.args.x = 3
  y();
  console.log(x); // foo.args.x = 2
}
foo(); // 3 2 2
console.log(x); // global.x = 1
```
