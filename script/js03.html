解构赋值 ​

数组的解构赋值 ​

js
let [foo, [[bar], baz]] = [1, [[2], 3]];
console.log(foo); // 1
console.log(bar); // 2
console.log(baz); // 3

let [, , third] = ["foo", "bar", "baz"];
console.log(third); // "baz"

let [first, , last] = [1, 2, 3];
console.log(first); // 1
console.log(last); // 3

let [head, ...tail] = [1, 2, 3, 4];
console.log(head); // 1
console.log(tail); // [2, 3, 4]

let [x, y, ...rest] = ["a"];
console.log(x); // "a"
console.log(y); // undefined
console.log(rest); // []

所有实现了 Iterable 接口的对象, 都可以使用解构赋值

js
let [x, y, z] = new Set(["a", "b", "c"]);

function* fibs() {
  let curr = 0;
  let next = 1;
  while (true) {
    yield curr;
    [curr, next] = [next, curr + next];
  }
}

let [first, second, third] = fibs();
console.log(first, second, third); // 0 1 1

解构赋值可以指定默认值

js
let [foo = true] = [];
console.log(foo); // true

let [x, y = "b"] = ["a"];
console.log(x, y); // a b

let [x1, y1 = "b"] = ["a", undefined];
console.log(x1, y1); // a b

let [x2 = 1] = [undefined];
console.log(x2); // 1

let [x3 = 1] = [null];
console.log(x3); // null

function f() {
  return "what";
}
let [x4 = f()] = [1];
console.log(x4); // 1

let [x5 = 1, y5 = x5] = [];
console.log(x5, y5); // 1 1

let [x6 = 1, y6 = x6] = [2];
console.log(x6, y6); // 2 2

let [x7 = 1, y7 = x7] = [1, 2];
console.log(x7, y7); // 1 2

let [x8 = y8, y8 = 1] = [];
console.log(x8, y8); // ReferenceError: y8 is not defined

对象的解构赋值是以下的简写

js
let obj = { foo: "a", bar: "b" };
// let { foo, bar } = { foo: "a", bar: "b" };

// 对象的解构赋值是以下的简写
let { foo /* prop */: foo /* var */, bar /* prop */: bar /* var */ } = obj;
console.log(foo, bar); // a b

// 起 "别名"
let { foo /* prop */: f /* var */, bar /* prop */: b /* var */ } = obj;
console.log(f, b); // a b

let greet = {
  p: ["hello", { y: "world" }],
};

let {
  p: [x /* 先按下标解构 */, { y /* 再按属性名解构 */ }],
} = greet;
console.log(x); // hello
console.log(y); // world

let {
  p,
  p: [x1, { y: y1 }],
} = greet;
console.log(p, x1, y1);
// ["hello", { y: "world" }]
// hello
// world

console.log(second); // 2
console.log(third); // 3

  return [x, y];

  return [x, y];
}

console.log(move2({ x: 1 })); // [1, undefined]
