# 函数

### 参数默认值

```js
function log(x, y = "world") {
  console.log(x, y);
}

log("hello"); // hello world
log("hello", "typescript"); // hello typescript
```

```js
function foo(x) {
  let x = 1; // SyntaxError: Identifier 'x' has already been declared
}

function bar(x = 5) {
  const x = 2; // SyntaxError: Identifier 'x' has already been declared
}
```

**对于有默认值的参数, 推荐写在参数列表的尾部**

### 函数的 length 属性

length 属性: 返回参数数量 (不统计有默认值的参数和后续参数; 不统计 rest 剩余参数)

```js
// length 属性不统计有默认值的参数和后续参数
(function (a) {}).length; // 1
(function (a = 5) {}).length; // 0
(function (a, b, c = 5, d, e) {}).length; // 2

// length 属性不统计 rest 剩余参数
(function (...rest) {}).length; // 0
```

使用参数默认值, 指定某参数不能省略, 如果省略则抛出错误

```js
function f(
  arg = (() => {
    throw new Error("cannot omitted");
  })(),
) {
  console.log(arg);
}
f(); // Error: cannot omitted
```

设置参数默认值为 undefined, 指定该参数可以省略

```js
function foo(optional = undefined) {}
```

### rest 剩余参数

rest 剩余参数必须是参数列表的最后一个参数

```js
// 使用 arguments
function sort1(...nums) {
  return nums.sort();
}
console.log(sort1(1, 3, 2)); // [ 1, 2, 3 ]

// 使用 rest 剩余参数
function sort2() {
  return Array.from(arguments).sort();
}
console.log(sort2(1, 3, 2)); // [ 1, 2, 3 ]
```

### 严格模式

- 对于 ES5, 函数体中可以指定严格模式
- 对于 ES6
  - 函数体中的严格模式, 同时限制函数参数和函数体
  - 如果函数参数使用了参数默认值, 解构赋值, rest 剩余参数, 则函数体中不能指定严格模式
  - 解决 1: 全局指定严格模式
  - 解决 2: 使用无参数的立即执行函数表达式 IIFE

```js
// SyntaxError: Illegal 'use strict' directive in function with non-simple parameter list
function f1(a, b = a) {"use strict"} // 参数默认值
function f2({a, b}) {"use strict"} // 解构赋值
function f3(...a) {"use strict"} // rest 剩余参数
```

```js
const f4 = (function () {
  "use strict";
  return function (...a) {
    return a.sort();
  };
})();
console.log(f4(1, 3, 2)); // [ 1, 2, 3 ]
```

### 函数的 name 属性

```js
function foo() {}
console.log(foo.name); // foo

const bar = function baz() {};
console.log(bar.name); // baz

new Function().name; // "anonymous"

function wtf() {}
console.log(wtf.bind({} /* bind `this` = {} */).name); // "bound wtf"
```

### 箭头函数

```js
// {} 被解释为代码块, 需要使用 () 包裹
let arrow1 = arg => /* return */ { k: 1, v: arg }; // SyntaxError

let arrow2 = arg => /* return */ ({ k: 1, v: arg });
console.log(arrow2('wtf')); // { k: 1, v: 'wtf' }
```

```js
// 无返回值的单行箭头函数
let arrow3 = (...args) => void console.log(args);
arrow3(1, 2, 3); // [ 1, 2, 3 ]
```

1. 箭头函数 (lambda 表达式) 没有 this, this 被视为变量向外层查找
   - 箭头函数不能作为构造函数, 即不能对箭头函数使用 new 关键字
   - 箭头函数不能使用 call(), apply() 和 bind() 方法改变 this 指向
2. 箭头函数的函数体中没有 arguments 对象, 可以使用 rest 剩余参数
3. 箭头函数不能作为生成器函数 (GeneratorFunction), 即不能在箭头函数的函数体中使用 yield 关键字

```js
function Timer() {
  this.cnt1 = 0;
  this.cnt2 = 0;
  let timerId1 = setInterval(() => {
    // console.log(this); // Timer { cnt1: ?, cnt2: 0 }
    this /* Timer */.cnt1++;
  }, 1000);

  let timerId2 = setInterval(function () {
    // console.log(this); // Timeout { ... }
    this /* Timeout */.cnt2++;
  }, 1000);

  return { timer: this, timerId1, timerId2 };
}

const { timer, timerId1, timerId2 } = new Timer();

setTimeout(() => {
  console.log("cnt1:", timer.cnt1); // cnt1: 3
  clearInterval(timerId1);
}, 3100);

setTimeout(() => {
  console.log("cnt2:", timer.cnt2); // cnt2: 0
  clearInterval(timerId2);
}, 3100);
```

### 尾调用

```js
function f(x) {
  return g(x); // 尾调用: return 函数调用
}

function f(x) {
  let y = g(x);
  return y; // 不是尾调用
}

function f(x) {
  return g(x) + 1; // 不是尾调用
}

function f(x) {
  g(x); // 不是尾调用
  // return undefined
}
```

尾调用优化

```js
// 非尾递归
function fib1(n) {
  if (n <= 1) {
    return 1;
  }
  return fib1(n - 1) + fib1(n - 2);
}

// 尾递归优化
function fib2(n, curr = 1, next = 1) {
  if (n <= 1) {
    return 1;
  }
  return fib2(n - 1, next, curr + next);
}
```
