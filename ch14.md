# Proxy

创建代理对象 `let op = new Proxy(target, handler)`

```js
let op = new Proxy(
  {}, // target 被代理对象
  {
    get: function (target, propKey, receiver) {
      // {} p {} false
      console.log(target, propKey, receiver, target === receiver);
      return Reflect.get(target, propKey, receiver);
    },
    set: function (target, propKey, value, receiver) {
      // { p: 1 } p { p: 1 } false
      console.log(target, propKey, receiver, target === receiver);
      return Reflect.set(target, propKey, value, receiver);
    },
  }, // handler 拦截器
);

op.attr = 1;
console.log(op.attr); // 1
```

```js
let op = new Proxy(
  {}, // target 被代理对象
  {
    get: function (target, propKey) {
      return 404;
    },
  }, // handler 拦截器
);

console.log(op.p); // 404
console.log(op.q); // 404
```

```js
let target = {};
let handler = {};
let op = new Proxy(target, handler);
op.p = "p2p";
console.log(target.p); // p2p
target.q = "qwq";
console.log(op.q); // qwq
```

Proxy 对象可以作为其他对象的原型对象

```js
let op = new Proxy(
  {}, // target 被代理对象
  {
    get: function (target, propKey) {
      return 404;
    },
  }, // handler 拦截器
);
let obj = Object.create(op /* prototyoe */);
// obj 对象上找不到 p 属性, 到原型对象上查找 p 属性, 导致被拦截
console.log(obj.p); // 404
```

一个拦截器可以设置多个拦截函数

```js
let handler = {
  get: function (target, name) {
    console.log("get", target, name);
    if (name === "prototype") {
      return Object.prototype;
    }
    return name;
  },

  apply: function (target, thisVal, args) {
    console.log("apply", target, thisVal, args);
    return args[0];
  },

  construct: function (target, args) {
    console.log("construct", target, args);
    return { value: args[1] };
  },
};

let fpo = new Proxy(
  function (x, y) {
    return x + y;
  }, // target
  handler,
);

fpo(1, 2); // apply
new fpo(3, 4); // construct
console.log(fpo.prototype /* get */ === Object.prototype); // true
console.log(fpo.foo /* get */ === "foo"); // true
```

### Proxy 实例方法

1. get() 拦截获取属性操作, 如果被代理对象的属性不可写 (writable) 且不可配置 (configurable), 则 get() 不能拦截获取属性操作
2. set() 拦截设置属性操作, 如果被代理对象的属性不可写 (writable), 则 set() 不能拦截设置属性操作
3. apply() 拦截方法的调用, 包括 (...args), apply(thisVal, arr), call(thisVal, ...args),
4. has() 拦截 "判断对象是否有某属性", 即 `propKey in proxyObj` 操作, 如果被代理对象的属性不可配置 (configurable) 或被代理对象禁止扩展, 则 has() 不能拦截 `propKey in proxyObj` 操作
5. construct() 拦截 new 关键字, 即拦截构造函数
6. deleteProperty() 拦截 delete 操作
7. defineProperty() 拦截 Object.defineProperty() 操作, 如果被代理对象禁止扩展, 则 defineProperty() 不能拦截 Object.defineProperty() 操作
8. getOwnPropertyDescriptor()
9. getPrototypeOf()
10. isExtensible()
11. ownKeys()
12. preventExtensions()
13. setPrototypeOf()

> get() 拦截获取属性操作

`get(target, propKey, receiver)`

- target: 被代理对象
- propKey: 属性名
- receiver: 当前 get 的对象, 通常是代理对象
- 返回代理对象

```js
let proto = new Proxy(
  {}, // target
  {
    get(target, propKey, receiver) {
      console.log("getter:", propKey);
      return target[propKey];
    },
  }, // handler
);

let obj = Object.create(proto);
// get 方法可以继承
console.log(obj.foo); // getter: foo, undefined
```

demo1: 创建支持负索引的数组 (的代理对象)

```js
function arrProxy(...items) {
  let handler = {
    get(target, propKey, receiver) {
      let idx = Number(propKey);
      if (idx < 0) {
        propKey = String(target.length + idx);
      }
      return Reflect.get(target, propKey, receiver);
    },
  };
  return new Proxy(items, handler);
}

let arr /* 代理对象 */ = arrProxy(1, 2, 3);
console.log(arr[-1]); // 3
```

如果被代理对象的属性不可写 (writable) 且不可配置 (configurable), 则 get() 不能拦截获取属性操作

```js
const target = Object.defineProperties(
  {},
  {
    foo: {
      value: "foo",
      writable: false, // 不可写
      configurable: false, // 不可配置
    },
  },
);

const handler = {
  get(target, propKey) {
    // return target[propKey];
    return "bar";
  },
};

let op = new Proxy(target, handler);
console.log(op.foo); // TypeError
```

> set() 拦截设置属性操作

`set(target, propKey, propValue, receiver)`

- target: 被代理对象
- propKey: 属性名
- propValue: 属性值
- receiver: 当前 get 的对象, 通常是代理对象
- 返回一个布尔值, 表示 set 成功或失败

```js
let validator = {
  set: function (target, propKey, propValue, receiver) {
    if (propKey === "age") {
      if (!Number.isInteger(propValue) || propValue < 0 || propValue > 150) {
        throw new RangeError("The age is invalid");
      }
    }
    target[propKey] = propValue;
    return true;
  },
};

let op = new Proxy({}, validator);
op.age = 3.5;
op.age = 666; // RangeError: The age is invalid
```

通常, 属性名以下划线开头, 表示内部属性

```js
let handler = {
  get(target, key) {
    if (key[0] === "_") {
      throw "Illegal";
    }
    return target[key];
  },
  set(target, key, value) {
    if (key[0] === "_") {
      throw "Illegal";
    }
    target[key] = value;
    return true;
  },
};

const op = new Proxy({}, handler);
try {
  console.log(op._prop);
} catch (what) {
  console.log(what); // Illegal
}

try {
  op._prop = "val";
} catch (what) {
  console.log(what); // Illegal
}
```

```js
let obj = {};
Object.defineProperty(obj, "foo", {
  value: "foo",
  writable: false,
});

let handler = {
  set: function (target, key, val, recv) {
    target[key] = val;
    return true;
  },
};

proxy.foo = "wtf"; // TypeError
```

> apply() 拦截方法的调用, 包括 (...args), apply(thisVal, arr), call(thisVal, ...args)

`apply(target, ctx, args)`

- target 被代理函数
- ctx 上下文对象, 即 this
- args 参数数组

```js
let target = function () {
  return "I am target";
};

let handler = {
  apply: function () {
    return "I am proxy";
  },
};

let op = new Proxy(target, handler);
console.log(op()); // I am proxy
```

```js
let handler = {
  apply(target, ctx, args) {
    console.log(arguments);
    // 0: [Function: target]
    // 1: undefined, null
    // 2. [1, 2]
    let [_target, thisVal, _args] = arguments;
    return Reflect.apply(_target, thisVal, _args) * 2;
  },
};

function target(a, b) {
  return a + b;
}

let op = new Proxy(target, handler);
console.log(op(1, 2)); // 6
console.log(op.call(null, 1, 2)); // 6
console.log(op.apply(null, [1, 2])); // 6

// 直接调用 Reflect.apply 方法 (target = op), 也会被拦截
console.log(Reflect.apply(op, null, [1, 2])); // 6
```

> has() 拦截 "判断对象是否有某属性", 即 `propKey in proxyObj` 操作

```js
let handler = {
  has(target, key) {
    if (key[0] === "_") {
      return false;
    }
    return key in target;
  },
};
let target = { _foo: "foo", foo: "foo" };
let op = new Proxy(target, handler);
console.log("foo" in op);
console.log("_foo" in op);
```

如果被代理对象的属性不可配置 (configurable) 或被代理对象禁止扩展, 则 has() 不能拦截 `propKey in proxyObj` 操作

```js
let target = { foo: 1 };
Object.preventExtensions(target);
let op = new Proxy(target, {
  has: function (target, key) {
    return false;
  },
});

console.log("foo" in op); // TypeError
```

has() 不拦截 for...in 循环

```js
let stu = { name: "foo", age: 3.5 };
let handler = {
  has(target, key) {
    if (key === "age") {
      return false;
    }
    return key in target;
  },
};
let op = new Proxy(stu, handler);
// 拦截 `propKey in proxyObj` 操作
console.log("age" in op); // false;

// has() 不拦截 for...in 循环
for (let key in op) {
  console.log(key); // name age
}
```

> construct() 拦截 new 关键字, 即拦截构造函数

`construct(target, args, newTarget)`

- target 被代理的 (被拦截的) 构造器
- args 构造器的参数数组
- newTarget 代理构造器
- 必须返回一个对象

```js
let Op = new Proxy(function () {}, {
  construct: function (target, args) {
    console.log(this === handler); // true
    return { foo: args };
    // return 1; // TypeError
  },
});

console.log(new Op(1, 2, 3)); // { foo: [ 1, 2, 3 ] }
```

> deleteProperty() 拦截 delete 操作

```js
let handler = {
  deleteProperty(target, key) {
    if (key[0] === "_") {
      throw new Error("Illegal");
    }
    delete target[key];
    return true;
  },
};

let target = { _foo: 1 };
let op = new Proxy(target, handler);
delete op._foo; // Error: Illegal
```

> defineProperty() 拦截 Object.defineProperty() 操作

如果被代理对象禁止扩展, 则 defineProperty() 不能拦截 Object.defineProperty() 操作

```js
let handler = {
  defineProperty(target, propKey, propDesc) {
    // {} foo { value: 1, writable: true, enumerable: true, configurable: true }
    console.log(target, propKey, propDesc);
    return false; // 返回一个布尔值, 表示定义属性有效或无效
  },
};
let target = {};
let op = new Proxy(target, handler);
op.foo = 1; // 定义属性无效
console.log(target, op); // {} {}
```
