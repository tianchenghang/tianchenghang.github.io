# Proxy

创建代理对象 `let proxyObj = new Proxy(target, handler)`

```js
let proxyObj = new Proxy(
  {}, // target
  {
    get: function (target, propKey, receiver) {
      // {} {} false
      console.log(target, receiver, target === receiver);
      return Reflect.get(target, propKey, receiver);
    },
    set: function (target, propKey, value, receiver) {
      // { p: 1 } { p: 1 } false
      console.log(target, receiver, target === receiver);
      return Reflect.set(target, propKey, value, receiver);
    },
  }, // handler
);

proxyObj.p = 1;
console.log(proxyObj.p); // 1
```

```js
let proxyObj = new Proxy(
  {}, // target
  {
    get: function (target, propKey) {
      return 416;
    },
  }, // handler
);

console.log(proxyObj.p1); // 416
console.log(proxyObj.p2); // 416
```

```js
let target = {};
let handler = {};
let proxyObj = new Proxy(target, handler);
proxyObj.p = "p2p";
console.log(target.p); // p2p
target.q = "qwq"
console.log(proxyObj.q); // qwq
```
