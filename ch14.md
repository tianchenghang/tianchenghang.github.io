# Proxy

创建代理对象 `let po = new Proxy(target, handler)`

```js
let po = new Proxy(
  {}, // target 被代理对象
  {
    get: function (target, propKey, receiver) {
      // {} p {} false
      console.log(target, propKey, receiver, target === receiver);
      return Reflect.get(target, propKey, receiver);
    },
    set: function (target, propKey, value, receiver) {
      // { p: 1 } p { p: 1 } false
      console.log(target, propKey, receiver, target === receiver);
      return Reflect.set(target, propKey, value, receiver);
    },
  }, // handler 拦截器
);

po.attr = 1;
console.log(po.attr); // 1
```

```js
let po = new Proxy(
  {}, // target 被代理对象

      return 404;
    },
  }, // handler 拦截器

```

```js
let target = {};
let handler = {};
let po = new Proxy(target, handler);
po.p = "p2p";

```

Proxy 对象可以作为其他对象的原型对象

```js
let po = new Proxy(
  {}, // target 被代理对象
  {
    get: function (target, propKey) {
      return 404;
    },
  }, // handler 拦截器
);
let obj = Object.create(po /* prototyoe */);
// obj 对象上找不到 p 属性, 到原型对象上查找 p 属性, 导致被拦截
console.log(obj.p); // 404
```

一个拦截器可以设置多个拦截函数

```js
let handler = {
  get: function (target, name) {
    console.log("get", target, name);
    if (name === "prototype") {
      return Object.prototype;
    }
    return name;
  },

  apply: function (target, thisVal, args) {
    console.log("apply", target, thisVal, args);
    return args[0];
  },

  construct: function (target, args) {
    console.log("construct", target, args);
    return { value: args[1] };
  },
};

let fpo = new Proxy(
  function (x, y) {
    return x + y;
  }, // target
  handler,
);

fpo(1, 2); // apply
new fpo(3, 4); // construct
console.log(fpo.prototype /* get */ === Object.prototype); // true
console.log(fpo.foo /* get */ === "foo"); // true
```

### Proxy 实例方法

> get() 拦截获取属性操作

`get(target, propKey, receiver)`

- target: 被代理对象
- propKey: 属性名
- receiver: 当前 get 的对象, 通常是代理对象
- 返回代理对象

```js
let proto = new Proxy(
  {}, // target
  {
    get(target, propKey, receiver) {
      console.log("getter:", propKey);
      return target[propKey];
    },
  }, // handler
);

let obj = Object.create(proto);
// get 方法可以继承
console.log(obj.foo); // getter: foo, undefined
```

demo1: 创建支持负索引的数组 (的代理对象)

```js
function arrProxy(...items) {
  let handler = {
    get(target, propKey, receiver) {
      let idx = Number(propKey);
      if (idx < 0) {
        propKey = String(target.length + idx);
      }
      return Reflect.get(target, propKey, receiver);
    },
  };
  return new Proxy(items, handler);
}

let arr /* 代理对象 */ = arrProxy(1, 2, 3);
console.log(arr[-1]); // 3
```

如果属性不可写 (writable) 且不可配置 (configurable) 则 proxy 对象不能拦截获取属性操作

```js
const target = Object.defineProperties(
  {},
  {
    foo: {
      value: "foo",
      writable: false, // 不可写
      configurable: false, // 不可配置
    },
  },
);

const handler = {
  get(target, propKey) {
    // return target[propKey];
    return "bar";
  },
};

let po = new Proxy(target, handler);
console.log(po.foo); // TypeError
```

> set() 拦截设置属性操作

`set(target, propKey, propValue, receiver)`

- target: 被代理对象
- propKey: 属性名
- propValue: 属性值
- receiver: 当前 get 的对象, 通常是代理对象
- 返回一个布尔值, 表示 set 成功或失败

```js
let validator = {
  set: function (target, propKey, propValue, receiver) {
    if (propKey === "age") {
      if (!Number.isInteger(propValue) || propValue < 0 || propValue > 150) {
        throw new RangeError("The age is invalid");
      }
    }
    target[propKey] = propValue;
    return true;
  },
};

let po = new Proxy({}, validator);
po.age = 3.5;
po.age = 666; // RangeError: The age is invalid
```

通常, 属性名以下划线开头, 表示内部属性

```js
let handler = {
  get(target, key) {
    if (key[0] === "_") {
      throw "Illegal";
    }
    return target[key];
  },
  set(target, key, value) {
    if (key[0] === "_") {
      throw "Illegal";
    }
    target[key] = value;
    return true;
  },
};

const po = new Proxy({}, handler);
try {
  console.log(po._prop);
} catch (what) {
  console.log(what); // Illegal
}

try {
  po._prop = "val";
} catch (what) {
  console.log(what); // Illegal
}
```

如果属性不可写 (writable) 则 proxy 对象不能拦截设置属性操作

```js
let obj = {};
Object.defineProperty(obj, "foo", {
  value: "foo",
  writable: false,
});

let handler = {
  set: function (tar, key, val, recv) {
    tar[key] = val;
    return true;
  },
};

let proxy = new Proxy(obj, handler);
proxy.foo = "wtf"; // TypeError
```

set 代理返回一个布尔值, 表示 set 成功或失败, 严格模式下, set 代理如果没有返回 true, 则报错
