# 对象

```js
const foo = "bar";

// 属性简写
const baz1 = { foo };
// 等价于
const baz2 = { foo: foo };
const baz3 = { ["f" + "oo"]: foo };
// { foo: 'bar' } { foo: 'bar' } { foo: 'bar' }
console.log(baz1, baz2, baz3);

// 方法简写
const obj1 = {
  meth() {
    return "I'm OK";
  },
};
// 等价于
const obj2 = {
  meth: function () {
    return "I'm OK";
  },
};
```

简写的方法不能作为构造函数

```js
const obj = {
  make() {
    this.foo = "bar";
  },
  Make: function () {
    this.foo = "bar";
  },
};
try {
  console.log(new obj.make());
} catch (e) {
  console.log(e); // TypeError: obj.make is not a constructor
}
console.log(new obj.Make()); // Make { foo: 'bar' }
```

### 属性表达式

```js
let obj = {};
obj.foo = 1;
// 使用属性表达式
obj["bar"] = 2;
console.log(obj); // { foo: 1, bar: 2 }
```

```js
let propKey = "prop key";
let obj = {
  foo: 1,
  bar: 2,
  // 使用 ["expr"] 属性表达式定义 field
  ["baz"]: 4,
  [propKey]: 3,
  // 使用 ["expr"] 属性表达式定义 method
  ["meth"]() {
    console.log("wtf");
  },
};

// { foo: 1, bar: 2, baz: 4, 'prop key': 3, meth: [Function: meth] }
console.log(obj);
obj.meth(); // wtf
```

属性表达式不能简写

```js
const foo1 = "abc";
const bar1 = { foo1 };
console.log(bar1); // { foo1: 'abc' }

// const foo2 = 'abc';
// const bar2 = { [foo2] }; // SyntaxError
// console.log(bar2)

const foo3 = "abc";
const bar3 = { [foo3]: "abc" };
console.log(bar3); // { abc: 'abc' }
```

如果表达式是一个对象, 则将对象转换为字符串 `[object Object]`

```js
const ka = { a: 1 };
const kb = { b: 2 };
const obj = {
  [ka]: "va",
  [kb]: "vb", // 覆盖 [ka]
};
console.log(obj); // { '[object Object]': 'vb' }
```

### 方法的 name 属性

```js
const person = {
  greet() {
    console.log("hello");
  },
};
console.log(person.greet.name); // greet
```

对于 setter, getter 方法

```js
const obj = {
  get foo() {
    return "foo";
  },
  set foo(x) {},
};

// 获取 obj 对象的 foo 属性的属性描述对象 pd
const pd = Object.getOwnPropertyDescriptor(obj, "foo");
console.log(pd.get.name); // 'get foo';
console.log(pd.set.name); // 'set foo';
```

其他

```js
console.log(new Function().name); // "anonymous"
console.log((() => {}).name); // undefined
console.log(function () {}.name); // undefined

let foo = new Function();
console.log(foo.name); // anonymous
let bar = function () {};
console.log(bar.name); // bar
```

### 属性的可枚举性

对象的每个属性都有一个属性描述对象 (Property Descriptor)

```js
let obj = { foo: 666 };
let pd = Object.getOwnPropertyDescriptor(obj, "foo");
// { value: 666, writable: true, enumerable: true, configurable: true }
console.log(pd);
```

以下操作会忽略 enumerable 为 false 的属性

- for...in 忽略 enumerable 为 false 的属性
- Object.keys(); JSON.stringify(); Object.assign() 忽略继承的属性, 忽略 enumerable 为 false 的属性

```js
// 数组的 length 属性是不可枚举的
console.log(Object.getOwnPropertyDescriptor([], "length").enumerable); // false
// 对象的 toString 方法是不可枚举的
console.log(
  Object.getOwnPropertyDescriptor(Object.prototype, "toString").enumerable,
); // false

class Foo {
  field1 = 1;
  field2 = 2;
  meth() {}
}
// ['constructor', 'meth']
console.log(Object.getOwnPropertyNames(Foo.prototype));
// 类的 prototype 对象的方法是不可枚举的
console.log(Object.getOwnPropertyDescriptor(Foo.prototype, "meth"));
// {
//   value: [Function: meth],
//   writable: true,
//   enumerable: false,
//   configurable: true
// }

let foo = new Foo();
for (let k in foo) {
  console.log(k); // field1 field2
}
```

### 属性的遍历

| 属性的遍历                        | 自身属性 | 继承属性 | 数值, 字符串属性 | Symbol 属性 | 可枚举性         |
| --------------------------------- | -------- | -------- | ---------------- | ----------- | ---------------- |
| for...in                          | 包括     | 包括     | 包括             | 不包括      | 可枚举           |
| Object.keys()                     | 包括     | 不包括   | 包括             | 不包括      | 可枚举           |
| Object.getOwnPropertyNames(obj)   | 包括     | 不包括   | 包括             | 不包括      | 可枚举, 不可枚举 |
| Object.getOwnPropertySymbols(obj) | 包括     | 不包括   | 不包括           | 包括        | 可枚举, 不可枚举 |
| Reflect.ownKeys(obj);             | 包括     | 不包括   | 包括             | 包括        | 可枚举, 不可枚举 |

1. 先遍历所有数值键，按数值升序排序
2. 再遍历所有字符串键
3. 后遍历所有 Symbol 键

### super 关键字

super 指向当前对象的原型对象, 只能出现在实例方法的方法体中

```js
const pa = {
  foo: "hello",
  bar() {
    console.log(this.foo);
  },
};

const son = {
  foo: "world",
  find() {
    console.log(super.foo === Object.getPrototypeOf(this).foo); // true
    super.bar(); // world
    debugger; // super.bar() 打印 world, 因为 this === son
    this.bar(); // world
    Object.getPrototypeOf(this).bar.call(this); // world
    return super.foo;
  },
};

// son --> pa
Object.setPrototypeOf(son, pa);
console.log(son.__proto__ === pa); // true
console.log(Object.getPrototypeOf(son) === pa); // true
console.log(son.find()); // "hello"
```

### 使用扩展运算符的解构赋值

```js
let { x, y, ...z } = { x: 1, y: 2, a: 3, b: 4 };
console.log(x, y, z); // 1 2 {a: 3, b: 4}

let { ...x } = null; // TypeError: Cannot destructure 'null'...
let { ...x } = undefined; // TypeError: Cannot destructure 'undefined'...
```

1. 未使用扩展运算符时, 可以解构赋值原型对象的属性 (继承的属性)
2. 使用扩展运算符时, 不能解构赋值原型对象的属性 (继承的属性)
3. 扩展运算符后必须接变量名, 不允许接解构赋值表达式

```js
let pa = { a: 1 };
let son = { b: 1 };
son.__proto__ = pa;
let { ...obj } = son;
console.log(obj, obj.a); // { b: 1 } undefined
```

```js
const obj = Object.create({ x: 1, y: 2 });
obj.z = 3;
// x: 未使用扩展运算符的解构赋值
// ...neo: 使用扩展运算符的解构赋值
let { x, ...neo } = obj; // 1
let { y, z } = neo; // undefined 2
console.log(x, y, z);

// SyntaxError: ... must be followed by an identifier in declaration contexts
let { a, ...{ b, c } } = obj;
```

### 扩展运算符

扩展运算符 ... 将被扩展对象的所有可遍历属性，拷贝到当前对象中

```js
let foo = { ...["a", "b", "c"] };
// {'0': 'a', '1': 'b', '2': 'c'}
console.log(foo);

console.log({ ...{}, a: 1 }); // {a: 1}

// 如果扩展运算符后不是对象, 则自动转换为对象
console.log({ ...1 }); // 1 自动转换为 Number{1}

// TODO
console.log({ ..."wtf" }); // {'0': 'w', '1': 't', '2': 'f'}
```
