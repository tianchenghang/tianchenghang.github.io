# Generator

- yield 是暂停标志, 只能在生成器函数中使用
- 生成器函数是惰性执行的, 即 yield 后的表达式是惰性求值的

```js
function* genFunc1() {
  yield 1;
  yield 2;
}
let gen1 = genFunc1();
console.log(gen1.next()); // {value: 1, done: false}
console.log(gen1.next()); // {value: 2, done: false}
// 生成器函数没有返回值
console.log(gen1.next()); // {value: undefined, done: true}

function* genFunc2() {
  yield 1;
  yield 2;
  return 3;
}
let gen2 = genFunc2();
console.log(gen2.next()); // {value: 1, done: false}
console.log(gen2.next()); // {value: 2, done: false}
// 生成器函数有返回值
console.log(gen2.next()); // {value: 3, done: true}
```

```js
function* genFunc() {
  console.log("3s 后才执行");
}

let gen = genFunc();
setTimeout(function () {
  gen.next();
}, 3000);
```

```js
let obj1 = {
  // 有 [Symbol.iterator] 方法, 实现了 Iterable 接口
  [Symbol.iterator]() {
    return {
      next() {
        return { value: 1, done: false };
      },
    };
  },
};
// for (let item of obj1) {
//   console.log(item);
// }

// 等价于
let obj2 /* Iterable */ = {
  // 有 [Symbol.iterator] 方法, 实现了 Iterable 接口
  *[Symbol.iterator]() {
    while (true) {
      yield 1;
    }
  },
};
// for (let item of obj2) {
//   console.log(item);
// }
```

生成器函数 GeneratorFunction 返回一个生成器 Generator 对象, 生成器对象实现了 IterableIterator 接口, generator[Symbol.iterator]() === generator

```ts
interface IterableIterator extends Iterator {
  [Symbol.iterator](): IterableIterator;
}
```

```js
let gen /* 生成器 Generator */ = genFunc /* 生成器函数 GeneratorFunction */();
console.log(gen[Symbol.iterator]() === gen);
```

### 生成器 Generator 对象的 next 方法

```js
function* genFunc() {
  for (let i = 0; true; i++) {
    let reset = yield i;
    console.log(reset);
    if (reset) {
      i = -1;
    }
  }
}
let gen = genFunc();
console.log(gen.next()); // undefined { value: 0, done: false }
console.log(gen.next()); // undefined { value: 1, done: false }
console.log(gen.next()); // true { value: 2, done: false }
// next 方法可以接受一个参数, 作为上一个 yield 表达式的返回值
console.log(gen.next(true /* reset */)); // { value: 0, done: false }
```

next 方法可以接受一个参数, 作为**上一个** yield 表达式的返回值

```js
function* genFunc1(x) {
  let y = yield x;
  let z = yield y;
  return x + y + z;
}
let gen1 = genFunc1(1);
console.log(gen1.next()); // { value: 1, done: false }
console.log(gen1.next()); // { value: undefined, done: false }
console.log(gen1.next()); // { value: NaN, done: true }

function* genFunc2(x) {
  let y = yield x; // y = 2;
  let z = yield y; // z = 3;
  return z; // 3
}
let gen2 = genFunc2(1);
console.log(gen2.next()); // { value: 1, done: false }
console.log(gen2.next(2)); // { value: 2, done: false }
console.log(gen2.next(3)); // { value: 3, done: true }
```

使用生成器函数 GeneratorFunction 和 for...of 遍历, 实现斐波那契数列

```js
function* fib() {
  let cur = 0,
    next = 1;
  while (true) {
    yield cur;
    [cur, next] = [next, cur + next];
  }
}

// 使用 for...of 遍历时, 不需要调用 next 方法
for (let item of fib()) {
  if (item > 100) {
    break;
  }
  console.log(item);
}
```

使用生成器函数 GeneratorFunction 实现可 for...of 遍历的对象

```js
let obj = { foo: 1, bar: 2 };

function* parseEntries(obj) {
  let propKeys = Reflect.ownKeys(obj);
  for (let propKey /* String | Symbol */ of propKeys) {
    yield [propKey, obj[propKey]];
  }
}
for (let [key, val] of parseEntries(obj)) {
  // foo => 1
  // bar => 2
  console.log(key, "=>", val);
}

obj[Symbol.iterator] = function* () {
  let propKeys = Object.keys(this);
  for (let propKey /* String */ of propKeys) {
    yield [propKey, obj[propKey]];
  }
};

for (let [key, val] of obj) {
  // foo => 1
  // bar => 2
  console.log(key, "=>", val);
}

// 等价于
obj[Symbol.iterator] = function () {
  let _this = this;
  let propKeys = Object.keys(this);
  let idx = 0;
  return {
    next() {
      if (idx < this.length) {
        return {
          value: _this[propKeys[idx++]],
          done: false,
        };
      }
      return { done: true };
    },
  };
};

for (let [key, val] of obj) {
  // foo => 1
  // bar => 2
  console.log(key, "=>", val);
}
```
